\section{常用公式、定理与结论}

\subsection{集合与常用逻辑用语}

\begin{itemize}
  \item \textbf{集合运算}：$A \cap B = \{x \mid x \in A \text{ 且 } x \in B\}$，$A \cup B = \{x \mid x \in A \text{ 或 } x \in B\}$
  \item \textbf{补集}：$\complement_U A = \{x \mid x \in U \text{ 且 } x \notin A\}$
  \item \textbf{摩根律}：$\complement_U(A \cap B) = \complement_U A \cup \complement_U B$，$\complement_U(A \cup B) = \complement_U A \cap \complement_U B$
  \item \textbf{子集个数}：含 $n$ 个元素的集合有 $2^n$ 个子集，$2^n-1$ 个真子集，$2^n-2$ 个非空真子集
  \item \textbf{量词否定}：$\forall x, p(x)$ 否定为 $\exists x, \lnot p(x)$；$\exists x, p(x)$ 否定为 $\forall x, \lnot p(x)$
  \item \textbf{充分必要条件}：$p \Rightarrow q$ 则 $p$ 是 $q$ 的充分条件，$q$ 是 $p$ 的必要条件
  \item \textbf{易错}：$\{0\}$ 与 $\varnothing$ 的区别；$0$ 与 $\{0\}$ 的区别；元素与集合用 $\in$，集合与集合用 $\subseteq$
\end{itemize}

\subsection{复数}

\[
z = a + bi,\qquad i^2 = -1,\qquad \bar{z} = a - bi,\qquad |z| = \sqrt{a^2 + b^2}
\]

\textbf{四则运算}：
\[
(a+bi) \pm (c+di) = (a\pm c) + (b\pm d)i
\]
\[
(a+bi)(c+di) = (ac-bd) + (ad+bc)i
\]
\[
\frac{a+bi}{c+di} = \frac{(a+bi)(c-di)}{c^2+d^2}
\]

\textbf{常用结论}：
\begin{itemize}
  \item $i$ 的幂：$i^1 = i,\; i^2 = -1,\; i^3 = -i,\; i^4 = 1$，周期为 4
  \item $z \cdot \bar{z} = |z|^2$（分母实数化的关键）
  \item 实系数一元二次方程虚根成对出现（共轭）
  \item 复数 $z$ 在复平面内对应的点为 $(a,b)$，$|z|$ 为到原点距离
  \item $|z_1 - z_2|$ 表示复平面上两点间距离
\end{itemize}

\subsection{不等式}

\textbf{基本不等式链}（$a,b>0$）：
\[
\frac{2}{\frac{1}{a}+\frac{1}{b}} \le \sqrt{ab} \le \frac{a+b}{2} \le \sqrt{\frac{a^2+b^2}{2}}
\]
（调和平均 $\le$ 几何平均 $\le$ 算术平均 $\le$ 平方平均）

\textbf{常用变形}：
\[
a^2 + b^2 \ge 2ab,\quad \frac{b}{a} + \frac{a}{b} \ge 2\;(ab>0),\quad a + \frac{1}{a} \ge 2\;(a>0)
\]
\[
\frac{a^2}{b} \ge 2a - b\;(b>0),\quad \sqrt{a^2+b^2} \ge \frac{\sqrt{2}}{2}(a+b)
\]

\textbf{柯西不等式}：
\[
(a_1^2 + a_2^2)(b_1^2 + b_2^2) \ge (a_1b_1 + a_2b_2)^2
\]

\textbf{绝对值不等式}：
\[
|a| - |b| \le |a \pm b| \le |a| + |b|
\]

\textbf{重要结论}：
\begin{itemize}
  \item 一元二次不等式 $ax^2 + bx + c > 0$：$a>0$ 且 $\Delta<0$ 时恒成立
  \item $ax^2 + bx + c \ge 0$ 恒成立 $\Leftrightarrow$ $a>0$ 且 $\Delta \le 0$
  \item $|x-a| + |x-b|$ 的最小值为 $|a-b|$
  \item 线性规划中，目标函数 $z = ax+by$ 的最值在可行域顶点处取到
\end{itemize}

\textbf{函数周期性}——几个常用等价条件：
\begin{itemize}
  \item $f(x+a) = f(x) \Rightarrow T = a$
  \item $f(x+a) = -f(x) \Rightarrow T = 2a$
  \item $f(x+a) = \dfrac{1}{f(x)} \Rightarrow T = 2a$
  \item $f(x+a) = \dfrac{1+f(x)}{1-f(x)} \Rightarrow T = 4a$
\end{itemize}

\textbf{函数对称性}：
\begin{itemize}
  \item $f(a+x) = f(a-x) \Leftrightarrow$ 关于 $x=a$ 对称
  \item $f(a+x) + f(a-x) = 2b \Leftrightarrow$ 关于 $(a,b)$ 对称
  \item $f(x+a) = f(-x+b) \Leftrightarrow$ 关于 $x=\frac{a+b}{2}$ 对称
\end{itemize}

\textbf{奇偶性判定}：
\begin{itemize}
  \item 奇函数：$f(-x) = -f(x)$，图象关于原点对称
  \item 偶函数：$f(-x) = f(x)$，图象关于 $y$ 轴对称
  \item 奇函数 $f(0)=0$（若 $x=0$ 在定义域内）
  \item 常见奇函数：$\sin x,\;\tan x,\;x^{2n+1},\;\ln\frac{1-x}{1+x},\;\ln(x+\sqrt{1+x^2})$
  \item 常见偶函数：$\cos x,\;x^{2n},\;|x|$
\end{itemize}

\subsection{三角函数与解三角形}

\textbf{同角关系}：
\[
\sin^2\alpha + \cos^2\alpha = 1,\quad \tan\alpha = \frac{\sin\alpha}{\cos\alpha}
\]

\textbf{和差角公式}：
\[
\sin(\alpha\pm\beta) = \sin\alpha\cos\beta \pm \cos\alpha\sin\beta
\]
\[
\cos(\alpha\pm\beta) = \cos\alpha\cos\beta \mp \sin\alpha\sin\beta
\]
\[
\tan(\alpha\pm\beta) = \frac{\tan\alpha \pm \tan\beta}{1 \mp \tan\alpha\tan\beta}
\]

\textbf{二倍角公式}：
\[
\sin2\alpha = 2\sin\alpha\cos\alpha
\]
\[
\cos2\alpha = \cos^2\alpha - \sin^2\alpha = 2\cos^2\alpha - 1 = 1 - 2\sin^2\alpha
\]
\[
\tan2\alpha = \frac{2\tan\alpha}{1-\tan^2\alpha}
\]

\textbf{降幂公式}（高频）：
\[
\sin^2\alpha = \frac{1 - \cos2\alpha}{2},\quad \cos^2\alpha = \frac{1 + \cos2\alpha}{2},\quad \sin\alpha\cos\alpha = \frac12\sin2\alpha
\]

\textbf{辅助角公式}（高频）：
\[
a\sin x + b\cos x = \sqrt{a^2+b^2}\,\sin(x+\varphi),\quad \tan\varphi = \frac{b}{a}
\]

\textbf{诱导公式}：奇变偶不变，符号看象限。

\textbf{诱导口诀}：
\begin{center}
$\sin(\pi-\alpha)=\sin\alpha$,\; $\cos(\pi-\alpha)=-\cos\alpha$,\;
$\sin(\pi+\alpha)=-\sin\alpha$,\; $\cos(\pi+\alpha)=-\cos\alpha$
\end{center}

\textbf{解三角形}：
\[
\frac{a}{\sin A} = \frac{b}{\sin B} = \frac{c}{\sin C} = 2R
\]
\[
a^2 = b^2 + c^2 - 2bc\cos A,\quad
\cos A = \frac{b^2 + c^2 - a^2}{2bc}
\]
\[
S = \frac12 bc\sin A = \frac12 ac\sin B = \frac12 ab\sin C = \frac{abc}{4R} = \frac12 (a+b+c)r
\]

\textbf{重要结论}：
\begin{itemize}
  \item $A > B \iff a > b \iff \sin A > \sin B$
  \item $\sin(A+B) = \sin C$，$\cos(A+B) = -\cos C$
  \item 锐角 $\triangle ABC$ 中：$\sin A > \cos B$（因为 $A+B > \frac{\pi}{2}$）
  \item $b^2 = a^2 + c^2 \Leftrightarrow B = \frac{\pi}{2}$
  \item $b^2 < a^2 + c^2 \Leftrightarrow B < \frac{\pi}{2}$（锐角）
  \item $b^2 > a^2 + c^2 \Leftrightarrow B > \frac{\pi}{2}$（钝角）
\end{itemize}

\subsection{平面向量}

\[
\vec{a} = (x,y),\quad |\vec{a}| = \sqrt{x^2+y^2}
\]
\[
\vec{a}\cdot\vec{b} = x_1x_2 + y_1y_2 = |\vec{a}||\vec{b}|\cos\theta
\]
\[
\vec{a} \perp \vec{b} \iff \vec{a}\cdot\vec{b} = 0 \iff x_1x_2 + y_1y_2 = 0
\]
\[
\vec{a} \parallel \vec{b} \iff \vec{a} = \lambda\vec{b} \iff x_1y_2 - x_2y_1 = 0
\]
\[
\cos\theta = \frac{\vec{a}\cdot\vec{b}}{|\vec{a}||\vec{b}|},\quad
\text{投影：}|\vec{a}|\cos\theta = \frac{\vec{a}\cdot\vec{b}}{|\vec{b}|}
\]

\textbf{重要结论}：
\begin{itemize}
  \item $\vec{a}$ 在 $\vec{b}$ 上的投影向量：$\dfrac{\vec{a}\cdot\vec{b}}{|\vec{b}|^2}\,\vec{b}$
  \item $\overrightarrow{AB} = \vec{b} - \vec{a}$
  \item $M$ 为 $AB$ 中点 $\Rightarrow \overrightarrow{OM} = \dfrac12(\overrightarrow{OA}+\overrightarrow{OB})$
  \item $G$ 为 $\triangle ABC$ 重心 $\Rightarrow \overrightarrow{GA}+\overrightarrow{GB}+\overrightarrow{GC}=\vec{0}$，且 $G\left(\frac{x_1+x_2+x_3}{3},\frac{y_1+y_2+y_3}{3}\right)$
  \item $P$ 在直线 $AB$ 上 $\iff \overrightarrow{AP} = \lambda\overrightarrow{AB}$（向量共线定理）
  \item $|\vec{a}+\vec{b}|^2 = |\vec{a}|^2 + |\vec{b}|^2 + 2\vec{a}\cdot\vec{b}$
  \item $|\vec{a}-\vec{b}|^2 = |\vec{a}|^2 + |\vec{b}|^2 - 2\vec{a}\cdot\vec{b}$
\end{itemize}

\begin{itemize}

  \item 面面平行：$a,b\subset\alpha,\;a\cap b=P,\;a\parallel\beta,\;b\parallel\beta \Rightarrow \alpha\parallel\beta$
