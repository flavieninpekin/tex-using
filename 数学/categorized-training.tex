\section{分类训练和限时训练}

\subsection{使用说明}

\begin{enumerate}
  \item \textbf{分类训练}：按知识点逐项突破，每组题做完后总结错因
  \item \textbf{限时训练}：模拟考试节奏，严格控制时间
  \item 每道题后标注了\ （★ 基础 / ★★ 中档 / ★★★ 拔高）\ 难度
\end{enumerate}

\subsection{一、分类训练}

\subsubsection{三角恒等变换与解三角形}

\begin{enumerate}
  \item[★] 已知 $\sin\alpha = \frac35$，$\alpha\in(\frac{\pi}{2},\pi)$，求 $\cos2\alpha$ 和 $\tan(\alpha+\frac{\pi}{4})$ 的值。
  \vspace{1cm}

  \item[★] 在 $\triangle ABC$ 中，$a=7$，$b=8$，$\cos B = -\frac17$，求 $\angle A$。
  \vspace{1cm}

  \item[★★] 已知函数 $f(x) = \sin2x + \sqrt{3}\cos2x$，求 $f(x)$ 的单调递增区间和对称轴方程。
  \vspace{1cm}

  \item[★★] 在 $\triangle ABC$ 中，$a,b,c$ 分别为角 $A,B,C$ 的对边，$2b\cos C = 2a - c$。
  （1）求 $B$；（2）若 $b=2\sqrt{3}$，求 $\triangle ABC$ 面积的最大值。
  \vspace{2cm}
\end{enumerate}

\subsubsection{数列}

\begin{enumerate}
  \item[★] 等差数列 $\{a_n\}$ 中，$a_1=1$，$a_3=5$，求 $S_{10}$。
  \vspace{1cm}

  \item[★] 等比数列 $\{a_n\}$ 中，$a_1=2$，$a_4=16$，求 $a_n$ 和 $S_5$。
  \vspace{1cm}

  \item[★★] 已知数列 $\{a_n\}$ 满足 $a_1=1$，$a_{n+1}=2a_n+1$，求 $\{a_n\}$ 的通项公式。
  （提示：构造 $a_{n+1}+1 = 2(a_n+1)$）
  \vspace{1.5cm}

  \item[★★] 已知 $S_n$ 是数列 $\{a_n\}$ 的前 $n$ 项和，$a_1=1$，$S_{n+1}=4a_n+2$。
  （1）设 $b_n = a_{n+1} - 2a_n$，求证 $\{b_n\}$ 是等比数列；
  （2）求 $\{a_n\}$ 的通项公式。
  \vspace{2cm}
\end{enumerate}

\subsubsection{立体几何}

\begin{enumerate}
  \item[★] 正方体 $ABCD-A_1B_1C_1D_1$ 中，求异面直线 $A_1B$ 与 $B_1C$ 所成角的大小。
  \vspace{1cm}

  \item[★★] 三棱锥 $P-ABC$ 中，$PA\perp$ 平面 $ABC$，$\angle ABC=90^\circ$，$PA=AB=BC=2$。
  （1）求证：$BC\perp$ 平面 $PAB$；
  （2）求直线 $PC$ 与平面 $PAB$ 所成角的正弦值。
  \vspace{2cm}

  \item[★★] 四棱锥 $P-ABCD$ 中，底面 $ABCD$ 是矩形，$PA\perp$ 底面 $ABCD$，
  $PA=AD=2$，$AB=3$。$E$ 为 $PD$ 中点。
  （1）求二面角 $E-AC-B$ 的余弦值；
  （2）求点 $B$ 到平面 $AEC$ 的距离。
  \vspace{2cm}
\end{enumerate}

\subsubsection{概率与统计}

\begin{enumerate}
  \item[★] 袋中有 3 个红球、2 个白球，从中任取 2 个球。求：
  （1）恰好取到 1 个红球的概率；
  （2）取到红球个数的分布列和期望。
  \vspace{1.5cm}

  \item[★★] 某射击运动员每次命中概率为 $0.8$，独立射击 5 次。
  （1）求恰好命中 4 次的概率；
  （2）求命中次数的数学期望和方差。
  \vspace{1.5cm}

  \item[★★] 某产品质检，一级品率为 $0.9$。现从产品中随机抽取 $n$ 件进行检测，
  要使至少有一件一级品的概率不低于 $0.999$，求 $n$ 的最小值。
  （参考：$\lg 0.9 \approx -0.0458$）
  \vspace{1.5cm}
\end{enumerate}

\subsubsection{解析几何}

\begin{enumerate}
  \item[★] 已知椭圆 $C:\dfrac{x^2}{4}+\dfrac{y^2}{3}=1$，$F$ 为右焦点，过 $F$ 的直线 $l$ 与
  $C$ 交于 $A,B$ 两点。求 $\triangle AOB$ 面积的最大值。
  \vspace{2cm}

  \item[★★] 双曲线 $C:\dfrac{x^2}{a^2}-\dfrac{y^2}{b^2}=1\;(a>0,b>0)$ 的右焦点为 $F(2,0)$，
  渐近线方程为 $y=\pm\sqrt{3}x$。
  （1）求 $C$ 的方程；
  （2）过 $F$ 的直线与 $C$ 交于 $A,B$，$AB$ 中点为 $M$，$O$ 为原点，
  当 $OM\perp AB$ 时，求直线 $AB$ 的方程。
  \vspace{2cm}

  \item[★★★] 抛物线 $y^2=2px\;(p>0)$ 的焦点为 $F$，过 $F$ 的直线交抛物线于 $A,B$，
  且 $|AF|=2|BF|$，求直线 $AB$ 的斜率。
  \vspace{2cm}
\end{enumerate}

\subsubsection{函数与导数}

\begin{enumerate}
  \item[★] 求函数 $f(x)=x^3-3x^2+1$ 的单调区间和极值。
  \vspace{1.5cm}

  \item[★★] 已知函数 $f(x)=\ln x - a(x-1)$，$a\in\mathbb{R}$。讨论 $f(x)$ 的单调性。
  \vspace{2cm}

  \item[★★] 已知函数 $f(x)=e^x - ax - 1$。
  （1）当 $a=1$ 时，求 $f(x)$ 的最小值；
  （2）若 $f(x) \ge 0$ 恒成立，求 $a$ 的取值范围。
  \vspace{2cm}

  \item[★★★] 已知函数 $f(x)=x\ln x - ax^2$ 有两个极值点 $x_1,x_2\;(x_1<x_2)$。
  （1）求 $a$ 的取值范围；
  （2）求证：$x_1x_2 > 1$。
  \vspace{2.5cm}
\end{enumerate}

\subsection{二、限时训练}

\subsubsection{限时训练一（基础巩固，30 分钟）}

\begin{enumerate}
  \item（5 分）已知集合 $A=\{x\mid x^2-3x+2\le 0\}$，$B=\{x\mid 2^x\ge 2\}$，求 $A\cap B$。
  \item（5 分）计算 $\dfrac{1+2i}{3-4i}$。
  \item（5 分）$\triangle ABC$ 中，$a=2$，$b=3$，$\cos C = \frac14$，求 $c$。
  \item（5 分）等差数列 $\{a_n\}$ 中，$a_2=3$，$a_5=9$，求 $S_8$。
  \item（12 分）已知向量 $\vec{a}=(1,2)$，$\vec{b}=(-2,1)$。
  （1）求 $|\vec{a}+\vec{b}|$；
  （2）求 $\vec{a}$ 与 $\vec{b}$ 的夹角 $\theta$ 的余弦值。
  \item（12 分）函数 $f(x)=x^3-3x$，求 $f(x)$ 在 $[-2,2]$ 上的最大值和最小值。
\end{enumerate}

\textbf{建议}：30 分钟内完成，目标得分 40+/44。

\subsubsection{限时训练二（中档强化，45 分钟）}

\begin{enumerate}
  \item（5 分）函数 $f(x)=\sqrt{4-x^2}$ 与 $g(x)=\ln(x+1)$ 的定义域交集为\_\_\_\_\_\_。
  \item（5 分）$(1+2x)^6$ 展开式中 $x^3$ 的系数为\_\_\_\_\_\_。
  \item（5 分）正方体 $ABCD-A_1B_1C_1D_1$ 中，$M$ 为 $CC_1$ 中点，求 $BM$ 与底面 $ABCD$ 所成角的正切值。
  \item（12 分）已知函数 $f(x)=\sin2x+2\sqrt{3}\sin^2x$。
  （1）求 $f(x)$ 的最小正周期；
  （2）求 $f(x)$ 在 $[0,\frac{\pi}{2}]$ 上的值域。
  \item（12 分）数列 $\{a_n\}$ 中，$a_1=2$，$a_{n+1}=2a_n-1$。
  （1）求证 $\{a_n-1\}$ 是等比数列；
  （2）求 $\{a_n\}$ 的前 $n$ 项和 $S_n$。
  \item（12 分）甲乙两选手比赛，每局甲胜概率 $0.6$，乙胜概率 $0.4$，比赛采用 3 局 2 胜制。
  求甲获得比赛胜利的概率。
\end{enumerate}

\textbf{建议}：45 分钟内完成，目标得分 40+/51。

\subsubsection{限时训练三（综合拔高，60 分钟）}

\begin{enumerate}
  \item（5 分）已知 $f(x)=e^x-ax$ 在 $[0,1]$ 上单调递减，则 $a$ 的取值范围是\_\_\_\_\_\_。
  \item（5 分）过点 $A(1,2)$ 的直线与圆 $x^2+y^2=4$ 相切，则该直线的方程为\_\_\_\_\_\_。
  \item（5 分）$\triangle ABC$ 中，$\overrightarrow{AB}\cdot\overrightarrow{AC}=4$，$|\overrightarrow{AB}|=2$，$|\overrightarrow{AC}|=4$，则 $\triangle ABC$ 的面积为\_\_\_\_\_\_。
  \item（12 分）椭圆 $C:\dfrac{x^2}{a^2}+\dfrac{y^2}{b^2}=1\;(a>b>0)$ 离心率 $e=\frac{\sqrt{6}}{3}$，
  过右焦点 $F$ 的直线 $l$ 与 $C$ 交于 $A,B$，$|AB|$ 的最小值为 $4$。
  （1）求椭圆方程；
  （2）若 $OA\perp OB$，求直线 $l$ 的方程。
  \item（12 分）函数 $f(x)=\ln x - \frac12 x^2 + x$。
  （1）求 $f(x)$ 的单调区间；
  （2）求证：当 $x>1$ 时，$f(x) < \frac16(x^2-1)$。
\end{enumerate}

\textbf{建议}：60 分钟内完成，目标得分 30+/39。
