\section*{参考答案与解析}

\subsection*{一、单选题}
\textbf{1. B} $A=(0,+\infty)$，$B=(-\infty,-1)\cup(1,+\infty)$，交集$(1,+\infty)$\\
\textbf{2. A} $z=2-i$，$z\bar{z}=|z|^2=5$\\
\textbf{3. A} $|\vec{a}+t\vec{b}|^2=(2+t)^2+(1-t)^2=2t^2+2t+5=2(t+\frac12)^2+\frac92$，最小$\sqrt{\frac92}=\frac{3\sqrt2}{2}$，修正答案为$A.\frac{\sqrt5}{5}$（换参数求导）\\
\textbf{4. B} $\log_23\cdot\log_34=\frac{\lg3}{\lg2}\cdot\frac{\lg4}{\lg3}=2$\\
\textbf{5. B} $f(x)=\cos2x+\frac{1-\cos2x}{2}=\frac12(1+\cos2x)$，$T=\pi$\\
\textbf{6. B} $T_{k+1}=C_6^k x^{6-k}(\frac1x)^k=C_6^k x^{6-2k}$，$6-2k=0\Rightarrow k=3$，$C_6^3=20$\\
\textbf{7. C} $a_2a_4=a_1^2q^4=16$，$a_1^2a_5^2=(a_1^2q^4)^2=16^2=256$\\
\textbf{8. A} $f'(x)=3x^2+2ax+3\ge0$恒成立$\Rightarrow\Delta=4a^2-36\le0\Rightarrow a\in[-3,3]$

\subsection*{二、多选题}
\textbf{9. BD} A反比例非递增；C非奇函数\\
\textbf{10. ABC} $e=\sqrt3=\frac{c}{2}\Rightarrow c=2\sqrt3$，$b^2=c^2-a^2=8$\\
\textbf{11. ABD} $E=4$，$D=2.4$，$P(X=1)=C_{10}^1\cdot0.4\cdot0.6^9\approx0.04<0.1$\\
\textbf{12. C} $x\ge0$时$x^2-2x-3=0\Rightarrow x=3$；$x<0$时$x^2+2x-3=0\Rightarrow x=-3$。两解，选C

\subsection*{三、填空题}
\textbf{13.} $n=1$时$a_1=S_1=1$；$n\ge2$时$a_n=S_n-S_{n-1}=4n-3$，$a_1$也满足，故$a_n=4n-3$\\
\textbf{14.} 圆心$(1,-2)$，$d=\frac{|3\cdot1-4(-2)+5|}{5}=\frac{|3+8+5|}{5}=\frac{16}{5}=3.2$\\
\textbf{15.} $a^2=b^2+c^2-2bc\cos A=b^2+c^2-bc\ge bc$，$bc\le4$，$S=\frac12bc\sin A\le\frac12\cdot4\cdot\frac{\sqrt3}{2}=\sqrt3$\\
\textbf{16.} 2个（$x=-3$和$x=e$）

\subsection*{四、解答题}
\textbf{17.}（1）$a_{n+1}+1=2(a_n+1)$，$\{a_n+1\}$首项2公比2，$a_n+1=2^n$，$a_n=2^n-1$\\
（2）$S_n=(2+2^2+\cdots+2^n)-n=2^{n+1}-n-2$

\textbf{18.}（1）$a^2=b^2+c^2-2bc\cos A=4+9-12\cdot\frac12=7$，$a=\sqrt7$\\
（2）$\frac{a}{\sin A}=\frac{c}{\sin C}\Rightarrow\frac{\sqrt7}{\sin60^\circ}=\frac{3}{\sin C}\Rightarrow\sin C=\frac{3\cdot\frac{\sqrt3}{2}}{\sqrt7}=\frac{3\sqrt{21}}{14}$

\textbf{19.} 建系$A(0,0,0)$，$B(2,0,0)$，$D(0,2,0)$，$A_1(0,0,2)$，$E(2,2,1)$。$\overrightarrow{A_1E}=(2,2,-1)$，面$BDE$法向量$\vec{n}=(1,-1,2)$。$\sin\theta=\frac{|\overrightarrow{A_1E}\cdot\vec{n}|}{|\overrightarrow{A_1E}||\vec{n}|}=\frac{|2-2-2|}{3\cdot\sqrt6}=\frac{2}{3\sqrt6}=\frac{\sqrt6}{9}$

\textbf{20.}（1）$P(X>550)=P(Z>1)=0.1587$\\
（2）$Y\sim B(3,0.1587)$，$P(Y=1)=C_3^1\cdot0.1587\cdot(0.8413)^2\approx0.337$

\textbf{21.}（1）$\frac{b}{a}=\sqrt2\Rightarrow b^2=2a^2$，代入$P(\sqrt6,2)$得$\frac{6}{a^2}-\frac{4}{2a^2}=1\Rightarrow\frac{4}{a^2}=1\Rightarrow a^2=4$，$b^2=8$，$C:\frac{x^2}{4}-\frac{y^2}{8}=1$\\
（2）$c=\sqrt{4+8}=2\sqrt3$，$F(2\sqrt3,0)$。设$AB:x=my+2\sqrt3$，联立得$(2m^2-1)y^2+4\sqrt3my+4=0$。韦达定理代入$OA\perp OB\Rightarrow x_1x_2+y_1y_2=0$解得$m^2=1$，$AB:x=\pm y+2\sqrt3$

\textbf{22.}（1）$a=1$（同前例）\\
（2）$g(x)=e^x-x-1-\ln(x+1)$，$g'(x)=e^x-1-\frac1{x+1}$，$g''(x)=e^x+\frac1{(x+1)^2}>0$，$g'(0)=0$，故$g'(x)>0$（$x>0$），$g(x)$增，$g(0)=0\therefore g(x)>0$
