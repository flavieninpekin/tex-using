\section*{一、单选题（8×5=40分）}

\textbf{1.} $A=\{x\mid x^2-2x-3\le0\}$，$B=\{0,1,2,3,4\}$，$A\cap B=$
A. $\{0,1,2\}$ \quad B. $\{0,1,2,3\}$ \quad C. $\{1,2,3\}$ \quad D. $\{0,1,2,3,4\}$

\textbf{2.} 复数$z=\frac{3+i}{1-2i}$的模为
A. $2$ \quad B. $\sqrt2$ \quad C. $\sqrt5$ \quad D. $5$

\textbf{3.} 已知$\sin\alpha=\frac35$，$\alpha\in(\frac{\pi}{2},\pi)$，则$\cos(\alpha+\frac{\pi}{4})=$
A. $-\frac{7\sqrt2}{10}$ \quad B. $\frac{7\sqrt2}{10}$ \quad C. $-\frac{\sqrt2}{10}$ \quad D. $\frac{\sqrt2}{10}$

\textbf{4.} 函数$f(x)=x^2-2\ln x$的单调递减区间为
A. $(0,1)$ \quad B. $(-\infty,-1)\cup(0,1)$ \quad C. $(-\infty,-1)$ \quad D. $(-1,0)\cup(0,1)$

\textbf{5.} 已知$\vec{a}=(1,2)$，$\vec{b}=(2,k)$，若$\vec{a}$与$\vec{b}$的夹角为锐角，则$k$的范围是
A. $k>-1$ \quad B. $k>-1$且$k\neq4$ \quad C. $k<-1$ \quad D. $k>-1$且$k\neq-4$

\textbf{6.} 等差数列$\{a_n\}$中，$a_1+a_4+a_7=15$，则$a_4=$
A. $3$ \quad B. $5$ \quad C. $7$ \quad D. $9$

\textbf{7.} 过点$A(2,1)$与圆$x^2+y^2=5$相切的直线方程为
A. $2x+y=5$ \quad B. $x+2y=5$ \quad C. $2x+y-5=0$ \quad D. $x+2y-4=0$

\textbf{8.} 已知函数$f(x)=a^x+\log_a x\;(a>0,a\neq1)$在$[1,2]$上的最大值与最小值之和为$\log_a2+6$，则$a=$
A. $\frac12$ \quad B. $2$ \quad C. $3$ \quad D. $4$

\section*{四、解答题（共70分）}

\textbf{17.}（10分）等差数列$\{a_n\}$中$a_2=3$，$a_5=9$。（1）求通项；（2）设$b_n=\frac1{a_na_{n+1}}$，求$T_n$。

\textbf{18.}（12分）$\triangle ABC$中$\sin^2A-\sin^2B-\sin^2C=\sin B\sin C$。（1）求$A$；（2）若$a=2\sqrt3$，求$b+c$的最大值。

\textbf{19.}（12分）三棱锥$P-ABC$中$PA\perp$面$ABC$，$\angle ABC=90^\circ$，$PA=AB=BC=2$。（1）证$BC\perp$面$PAB$；（2）求$PC$与面$PAB$所成角的正弦值。

\textbf{20.}（12分）一个袋中装有4个白球、3个黑球，每次取1个，取后不放回，直到取到白球为止。$X$为取球次数。（1）求$X$的分布列；（2）求$E(X)$。

\textbf{21.}（12分）椭圆$C:\frac{x^2}{4}+\frac{y^2}{3}=1$，$A(-2,0),B(2,0)$。过$P(1,0)$的直线交$C$于$M,N$。$AM$与$BN$交于$Q$。求$Q$的轨迹方程。

\textbf{22.}（12分）$f(x)=\ln x - \frac12ax^2$。（1）讨论单调性；（2）若$f(x)$在$(0,+\infty)$上恒负，求$a$范围。

\section*{参考答案}

\textbf{1. B} $A=[-1,3]$，$A\cap B=\{0,1,2,3\}$\\
\textbf{2. B} $z=\frac{3+i}{1-2i}=\frac{(3+i)(1+2i)}{1+4}=\frac{1+7i}{5}$，$|z|=\frac{\sqrt{1+49}}{5}=\frac{5\sqrt2}{5}=\sqrt2$\\
\textbf{3. C} $\cos\alpha=-\frac45$，$\cos(\alpha+\frac{\pi}{4})=\frac{\sqrt2}{2}(\cos\alpha-\sin\alpha)=\frac{\sqrt2}{2}(-\frac45-\frac35)=-\frac{7\sqrt2}{10}$\\
\textbf{4. A} 定义域$(0,+\infty)$，$f'(x)=2x-\frac2x=\frac{2(x^2-1)}{x}<0\Rightarrow0<x<1$\\
\textbf{5. B} $\vec{a}\cdot\vec{b}=1\cdot2+2\cdot k=2+2k>0\Rightarrow k>-1$，且不共线$\Rightarrow k\neq4$\\
\textbf{6. B} $a_1+a_4+a_7=3a_4=15\Rightarrow a_4=5$\\
\textbf{7. A} 点$A$在圆上，切线$2x+1\cdot y=5\Rightarrow2x+y=5$\\
\textbf{8. B} 分类讨论$a>1$时最大$=a^2+\log_a2$，最小$=a+0$，合计$a^2+\log_a2+a=a^2+a+\log_a2=\log_a2+6\Rightarrow a^2+a=6\Rightarrow a=2$（$a>0$）\\

\textbf{9. BD} \textbf{10. AB} \textbf{11. BD} \textbf{12. ACD}\\

\textbf{13.} $[x^3+x^2]_0^1=1+1=2$\\
\textbf{14.} $p=2$，焦点到准线距离$p=2$\\
\textbf{15.} $c>0$，$0<a,b<1$，$c>1>a>b$，$c>a>b$\\
\textbf{16.} $a_n=1+\sum_{k=1}^{n-1}(2k+1)=1+(n-1)+(n-1)^2=n^2$\\

\textbf{17.} $d=2$，$a_n=2n-1$；$b_n=\frac1{(2n-1)(2n+1)}=\frac12(\frac1{2n-1}-\frac1{2n+1})$，$T_n=\frac12(1-\frac1{2n+1})=\frac{n}{2n+1}$\\
\textbf{18.} （1）$\sin^2A=\sin^2B+\sin^2C+\sin B\sin C$，正弦定理$a^2=b^2+c^2+bc$，$\cos A=\frac{b^2+c^2-a^2}{2bc}=-\frac12$，$A=120^\circ$\\
（2）$2R=\frac{a}{\sin A}=\frac{2\sqrt3}{\frac{\sqrt3}{2}}=4$，$b+c=2R(\sin B+\sin C)=4[\sin B+\sin(60^\circ-B)]=4\sqrt3\sin(B+30^\circ)\le4\sqrt3$\\
\textbf{19.} 略\\
\textbf{20.} $P(X=1)=\frac47$，$P(X=2)=\frac37\cdot\frac45=\frac{12}{35}$，$P(X=3)=\frac37\cdot\frac15\cdot\frac44=\frac3{35}$，$E(X)=1\cdot\frac47+2\cdot\frac{12}{35}+3\cdot\frac3{35}=1.4$\\
\textbf{21.} 略（参考例17，轨迹为$x=4$）\\
\textbf{22.} （1）$f'(x)=\frac1x-ax$，$a\le0$时$f'(x)>0$递增；$a>0$时$(0,\frac1{\sqrt a})$增$(\frac1{\sqrt a},+\infty)$减\\
（2）$f_{\max}=f(\frac1{\sqrt a})=\ln\frac1{\sqrt a}-\frac12<0\Rightarrow a>\frac1e$
