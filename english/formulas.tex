\subsection{高中英语核心语法要点}

\subsubsection{时态与语态（必考）}

\begin{center}
\begin{tabular}{c|c|c}
\textbf{时态} & \textbf{主动形式} & \textbf{被动形式} \\ \toprule
一般现在 & do/does & am/is/are done \\
一般过去 & did & was/were done \\
一般将来 & will do & will be done \\
现在进行 & am/is/are doing & am/is/are being done \\
过去进行 & was/were doing & was/were being done \\
现在完成 & have/has done & have/has been done \\
过去完成 & had done & had been done \\
过去将来 & would do & would be done \\
现在完成进行 & have/has been doing & -- \\
\bottomrule
\end{tabular}
\end{center}

\textbf{关键判断标志词}：
\begin{itemize}
  \item 一般现在：often, usually, always, sometimes, every day
  \item 一般过去：yesterday, last week, \dots ago, in 2020
  \item 现在完成：already, yet, just, ever, never, since, for, so far, recently
  \item 过去完成：by the time, before, after（两个动作先后）
\end{itemize}

\subsubsection{非谓语动词}

\textbf{不定式 to do}：
\begin{itemize}
  \item 作目的状语：\textit{I went there to see him.}
  \item 作宾语：want, hope, expect, decide, refuse, pretend
  \item 作宾补：tell, ask, allow, invite, force, enable
  \item 固定句型：It is + adj + to do / too...to / enough to
\end{itemize}

\textbf{动名词 doing}：
\begin{itemize}
  \item 作主语：\textit{Learning English is important.}
  \item 作宾语：enjoy, avoid, consider, suggest, miss, keep, practice, mind
  \item 介词后：\textit{look forward to doing, be used to doing, devote to doing}
\end{itemize}

\textbf{分词 doing/done}：
\begin{itemize}
  \item 主动进行用 doing：\textit{The boy sitting there is my brother.}
  \item 被动完成用 done：\textit{The book written by him is popular.}
  \item 独立主格：\textit{Weather permitting, we'll go.}
\end{itemize}

\subsubsection{三大从句}

\textbf{定语从句}：
\begin{itemize}
  \item 关系代词：who/whom(人), which(物), that(人+物), whose(所属)
  \item 关系副词：when(时间), where(地点), why(原因)
  \item 只用 that 不用 which 的情况：先行词有序数词、最高级、不定代词等
  \item 非限制性定语从句：用 which/as，不用 that
\end{itemize}

\textbf{名词性从句}（主语/宾语/表语/同位语从句）：
\begin{itemize}
  \item that：不充当成分，无意义（宾语从句可省略）
  \item whether/if：是否（whether 可放句首）
  \item wh- 词：what, which, who, when, where, why, how
\end{itemize}

\textbf{状语从句}：
\begin{itemize}
  \item 时间：when, while, as, before, after, since, until, as soon as
  \item 原因：because, since, as, now that
  \item 条件：if, unless, as long as, provided that
  \item 让步：although, though, even if, while, no matter + wh
  \item 目的：so that, in order that
  \item 结果：so...that, such...that
\end{itemize}

\subsubsection{特殊句式}

\textbf{强调句}：
\[
\text{It is/was + 被强调部分 + that/who + 剩余部分}
\]

\textbf{倒装句}：
\begin{itemize}
  \item 否定词位于句首：never, seldom, hardly, not only, little, nowhere
  \item only + 状语位于句首：\textit{Only then did I realize...}
  \item so/such...that 位于句首：\textit{So interesting is the book that...}
\end{itemize}

\textbf{虚拟语气}：
\begin{itemize}
  \item 与现在相反：if 过去式(were), would/could do
  \item 与过去相反：if had done, would/could have done
  \item 与将来相反：if should do/were to do, would do
  \item suggest/order/require that ... (should) do
  \item It is necessary/important that ... (should) do
  \item wish后的宾语从句：时态后退
\end{itemize}

\subsubsection{主谓一致}

\begin{itemize}
  \item and 连接两个主语 $\rightarrow$ 复数（同一概念时单数）
  \item either...or/neither...nor/not only...but also：就近原则
  \item as well as, together with, along with：就远原则
  \item 集合名词：class, family, team, group（强调整体单数，强调成员复数）
  \item 距离、金额、时间作主语 $\rightarrow$ 单数
  \item a number of + 复数，the number of + 单数
\end{itemize}

\subsection{常用写作句式}

\subsubsection{书信常用句型}

\begin{itemize}
  \item 开头：I am writing to invite / express / share / inform...
  \item 感谢：I would like to express my sincere gratitude for...
  \item 道歉：I am terribly sorry that... / Please accept my apology for...
  \item 建议：I'd like to suggest that... / It would be beneficial to...
  \item 请求：I would appreciate it if you could...
  \item 期待：I am looking forward to your early reply.
  \item 结尾：Best wishes / Yours sincerely / Looking forward to hearing from you.
\end{itemize}
